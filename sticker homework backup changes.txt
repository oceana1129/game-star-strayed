package edu.northeastern.numadfa24_group3;

import android.content.Intent;
import android.os.Bundle;
import android.util.Log;
import android.view.View;
import android.widget.EditText;
import android.widget.Toast;

import androidx.annotation.NonNull;
import androidx.appcompat.app.AppCompatActivity;
import androidx.core.graphics.Insets;
import androidx.core.view.ViewCompat;
import androidx.core.view.WindowInsetsCompat;

import com.google.firebase.database.DataSnapshot;
import com.google.firebase.database.DatabaseError;
import com.google.firebase.database.DatabaseReference;
import com.google.firebase.database.FirebaseDatabase;
import com.google.firebase.database.ValueEventListener;


public class FirebaseRealtimeDatabaseActivity extends AppCompatActivity {
    private static final String TAG = "FirebaseRealtimeDatabaseActivity___";
    private DatabaseReference userDatabase;
    private EditText usernameEditText;
    private User currUser;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        Log.v(TAG, "onCreate()");
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_firebase_realtime_database);
        userDatabase = FirebaseDatabase.getInstance().getReference("users");

        usernameEditText = findViewById(R.id.username_text_box);

        ViewCompat.setOnApplyWindowInsetsListener(findViewById(R.id.realtime_database), (v, insets) -> {
            Insets systemBars = insets.getInsets(WindowInsetsCompat.Type.systemBars());
            v.setPadding(systemBars.left, systemBars.top, systemBars.right, systemBars.bottom);
            return insets;
        });
    }

    private void saveUsername() {
        String username = usernameEditText.getText().toString().trim();

        if (!username.isEmpty()) {
            // Check if username already exists in Firebase
            userDatabase.orderByChild("username").equalTo(username).addListenerForSingleValueEvent(new ValueEventListener() {
                public void onDataChange(@NonNull DataSnapshot dataSnapshot) {
                    if (dataSnapshot.exists()) {
                        // Username already exists
                        Toast.makeText(FirebaseRealtimeDatabaseActivity.this, "Username already exists!", Toast.LENGTH_SHORT).show();
                    } else {
                        // Username does not exist, save the new user
                        String userId = userDatabase.push().getKey();
                        userDatabase.child(userId).child("username").setValue(username);
                        Toast.makeText(FirebaseRealtimeDatabaseActivity.this, "Username saved", Toast.LENGTH_SHORT).show();
                        currUser = new User(username);
                        usernameEditText.setText("");
                    }
                }

                public void onCancelled(@NonNull DatabaseError databaseError) {
                    Toast.makeText(FirebaseRealtimeDatabaseActivity.this, "Error checking username: " + databaseError.getMessage(), Toast.LENGTH_SHORT).show();
                }
            });
        } else {
            Toast.makeText(this, "Please enter a username", Toast.LENGTH_SHORT).show();
        }
    }

    private void attemptLogin() {
        String username = usernameEditText.getText().toString().trim();

        if (!username.isEmpty()) {
            // Check if username already exists in Firebase
            userDatabase.orderByChild("username").equalTo(username).addListenerForSingleValueEvent(new ValueEventListener() {
                public void onDataChange(@NonNull DataSnapshot dataSnapshot) {
                    if (dataSnapshot.exists()) {
                        // Username already exists
                        Toast.makeText(FirebaseRealtimeDatabaseActivity.this, "Logged in as " + username, Toast.LENGTH_SHORT).show();
                        currUser = new User(username);
                        usernameEditText.setText("");

                        Intent intent = new Intent(FirebaseRealtimeDatabaseActivity.this, StickerMainActivity.class);
                        intent.putExtra("username", username);
                        startActivity(intent);

                        // Close the current activity
                        finish();

                    } else {
                        Toast.makeText(FirebaseRealtimeDatabaseActivity.this, "Username does not exist.", Toast.LENGTH_SHORT).show();
                    }
                }

                public void onCancelled(@NonNull DatabaseError databaseError) {
                    Toast.makeText(FirebaseRealtimeDatabaseActivity.this, "Error checking username: " + databaseError.getMessage(), Toast.LENGTH_SHORT).show();
                }
            });
        } else {
            Toast.makeText(this, "Please enter a username", Toast.LENGTH_SHORT).show();
        }
    }

    public void onClick(View view) {
        int id = view.getId();
        if (id == R.id.back_button) {
            finish();
        } else if (id == R.id.sign_up_btn) {
            saveUsername();
        } else if (id == R.id.log_in_btn) {
            attemptLogin();
        }
    }

    @Override
    protected void onPause() {
        Log.v(TAG, "onPause()");
        super.onPause();
    }

    @Override
    protected void onDestroy() {
        Log.v(TAG, "onDestroy()");
        super.onDestroy();
    }

    @Override
    protected void onResume() {
        Log.v(TAG, "onResume()");
        super.onResume();
    }

    @Override
    protected void onRestart() {
        Log.v(TAG, "onRestart()");
        super.onRestart();
    }

    @Override
    protected void onStart() {
        Log.v(TAG, "onStart()");
        super.onStart();
    }

    @Override
    protected void onStop() {
        Log.v(TAG, "onStop()");
        super.onStop();
    }
}



package edu.northeastern.numadfa24_group3;

import android.content.Context;
import android.view.LayoutInflater;
import android.view.View;
import android.view.ViewGroup;

import androidx.annotation.NonNull;
import androidx.recyclerview.widget.RecyclerView;

import java.text.SimpleDateFormat;
import java.util.Date;
import java.util.List;
import java.util.Locale;

public class StickerHistoryAdapter extends RecyclerView.Adapter<StickerHistoryViewHolder> {
    private final List<StickersSent> historyList;
    private final Context context;

    public StickerHistoryAdapter(List<StickersSent> historyList, Context context) {
        this.historyList = historyList;
        this.context = context;
    }

    @NonNull
    @Override
    public StickerHistoryViewHolder onCreateViewHolder(@NonNull ViewGroup parent, int viewType) {
        View view = LayoutInflater.from(context).inflate(R.layout.sticker_profile_item_history, parent, false);
        return new StickerHistoryViewHolder(view);
    }

    @Override
    public void onBindViewHolder(@NonNull StickerHistoryViewHolder holder, int position) {
        StickersSent stickersSent = historyList.get(position);
        // if default text is null
        holder.receiver.setText(stickersSent.getReceiver() != null ? stickersSent.getReceiver() : "Unknown user");

        // format date
        Date date = stickersSent.getTimeSent();
        if (date != null) {
            SimpleDateFormat sdf = new SimpleDateFormat("dd-MM-yyyy HH:mm", Locale.getDefault());
            String formattedDateTime = sdf.format(date);
            holder.date.setText(formattedDateTime);
        } else {
            holder.date.setText("Unknown date"); // if date is null
        }
    }

    @Override
    public int getItemCount() {
        return historyList.size();
    }
}

    private String recipientUsername;
    private String accountUsername;
    private static final String TAG = "StickerSendSearchFragment";

    public StickerSendSearchFragment() {
        // Required empty public constructor
    }

    public static StickerSendSearchFragment newInstance() {
        return new StickerSendSearchFragment();
    }

    @Override
    public void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        if (getArguments() != null) {
            accountUsername = getArguments().getString("accountUsername"); // get username
        }
        userDatabase = FirebaseDatabase.getInstance().getReference("users"); // initialize firebase
    }

    @Override
    public View onCreateView(@NonNull LayoutInflater inflater, @Nullable ViewGroup container, @Nullable Bundle savedInstanceState) {
        // inflate layout
        View view = inflater.inflate(R.layout.fragment_sticker_send_search, container, false);

        init(view); // initialize components
        setBtnOnClickListeners(); // set button click listeners

        return view;
    }

    /**
     * initialize the UI
     *
     * @param view the root view from onCreateView
     */
    private void init(View view) {
        usernameTextBox = view.findViewById(R.id.textbox_username);
        btnSearch = view.findViewById(R.id.btn_search);
        btnRecommendedUser1 = view.findViewById(R.id.sticker_recommended_1);
        btnRecommendedUser2 = view.findViewById(R.id.sticker_recommended_2);
        btnRecommendedUser3 = view.findViewById(R.id.sticker_recommended_3);
    }

    /**
     * set up button click listeners
     */
    private void setBtnOnClickListeners() {
        // search for a user
        btnSearch.setOnClickListener(v -> verifyUser(usernameTextBox.getText().toString().trim()));

        // button click on recommended users
        btnRecommendedUser1.setOnClickListener(v -> verifyUser("ange"));
        btnRecommendedUser2.setOnClickListener(v -> verifyUser("test"));
        btnRecommendedUser3.setOnClickListener(v -> verifyUser("test1"));
    }

    private void verifyUser(String username) {
        // see if the username or textbox is empty
        if (username.isEmpty()) {
            Toast.makeText(getContext(), "Please enter a username", Toast.LENGTH_SHORT).show();
            return;
        }

        // if user entered their username
        if (Objects.equals(accountUsername, username)) {
            Toast.makeText(getContext(), "Cannot send stickers to self", Toast.LENGTH_SHORT).show();
            return;
        }
        Log.d(TAG, "Verifying user: " + username);
        // see if username exists in the database
        userDatabase.orderByChild("username").equalTo(username).get().addOnCompleteListener(task -> {

            if (task.isSuccessful()) {
                if (task.getResult().exists()) {                    // username exists
                    navigateToStickerSelect(username);                  // proceed to next screen
                } else {                                            // username does not exist
                    Toast.makeText(getContext(), "User not found", Toast.LENGTH_SHORT).show();
                }
            } else {                                                // error handling
                Toast.makeText(getContext(), "Error checking username: " + task.getException().getMessage(), Toast.LENGTH_SHORT).show();
            }
        });
    }

    /**
     * navigate to next fragment to select a sticker with the searched username
     *
     * @param username the verified username to send a sticker to
     */
    private void navigateToStickerSelect(String username) {
        StickerSendSelectFragment selectFragment = StickerSendSelectFragment.newInstance(username);
        replaceFragment(selectFragment);
    }

    /**
     * dynamically load new fragments onto the main activity
     *
     * @param fragment fragment information loaded
     */
    private void replaceFragment(Fragment fragment) {
        // does fragment require username?
        if (accountUsername != null) {

    }
